---
title: "Update Module"
api: "PUT /modules/{module_id}"
---

Update an existing module's title, browser settings, and steps. This replaces all existing steps with the new set provided.

<ParamField path="module_id" type="integer" required>
  The ID of the module to update.
</ParamField>

<ParamField body="title" type="string">
  The updated module name.
</ParamField>

<ParamField body="viewport_width" type="integer">
  Browser viewport width in pixels.
</ParamField>

<ParamField body="viewport_height" type="integer">
  Browser viewport height in pixels.
</ParamField>

<ParamField body="browser_zoom" type="number">
  Browser zoom level.
</ParamField>

<ParamField body="steps" type="object[]">
  The new set of steps for the module. This replaces all existing steps.

  <Expandable title="Step object properties">
    <ParamField body="step_number" type="integer" required>
      The position of this step in the sequence (starting from 1).
    </ParamField>

    <ParamField body="type" type="string" required>
      The step type: `"act"`, `"assert"`, or `"api"`.
    </ParamField>

    <ParamField body="action" type="string" required>
      A natural language instruction for the step.
    </ParamField>

    <ParamField body="use_vision" type="boolean">
      Whether to enable vision for this step. Defaults to `false`.
    </ParamField>
  </Expandable>
</ParamField>

### Example request body

```json
{
  "title": "Updated Login Module",
  "viewport_width": 1900,
  "viewport_height": 1200,
  "browser_zoom": 1,
  "steps": [
    {
      "step_number": 1,
      "type": "act",
      "action": "Navigate to the updated login page",
      "use_vision": false
    }
  ]
}
```

<ResponseField name="message" type="string">
  A confirmation message (e.g. `"Module updated successfully"`).
</ResponseField>

<ResponseField name="module" type="object">
  The updated module object.

  <Expandable title="Properties">
    <ResponseField name="id" type="integer">
      The unique ID of the module.
    </ResponseField>

    <ResponseField name="title" type="string">
      The updated module name.
    </ResponseField>

    <ResponseField name="type" type="string">
      Always `"step_module"` for modules.
    </ResponseField>

    <ResponseField name="status" type="string">
      The module status.
    </ResponseField>

    <ResponseField name="test_steps" type="object[]">
      The updated list of steps.
    </ResponseField>

    <ResponseField name="test_blueprints_using_module" type="object[]">
      Tests referencing this module.
    </ResponseField>

    <ResponseField name="company_id" type="integer">
      Your company ID.
    </ResponseField>

    <ResponseField name="browser_zoom" type="number">
      The browser zoom level.
    </ResponseField>

    <ResponseField name="viewport_width" type="integer">
      Browser viewport width.
    </ResponseField>

    <ResponseField name="viewport_height" type="integer">
      Browser viewport height.
    </ResponseField>

    <ResponseField name="usage_count" type="integer">
      Number of tests using this module.
    </ResponseField>

    <ResponseField name="created_at" type="string">
      ISO 8601 timestamp of when the module was created.
    </ResponseField>

    <ResponseField name="updated_at" type="string">
      ISO 8601 timestamp of the last update.
    </ResponseField>
  </Expandable>
</ResponseField>

<Warning>
  Updating a module replaces **all** existing steps with the steps provided in the request body. Make sure to include all steps you want to keep.
</Warning>
